-- Creating the "Users" table
CREATE TABLE Users (
  ID SERIAL PRIMARY KEY,
  FirstName VARCHAR(255),
  LastName VARCHAR(255),
  Email VARCHAR(255),
  Password VARCHAR(255),
  Nickname VARCHAR(255)
);

-- Creating the "Posts" table
CREATE TABLE Posts (
  ID SERIAL PRIMARY KEY,
  User_ID INT REFERENCES Users(ID),
  Title VARCHAR(255),
  Content TEXT,
  CreationDate DATE
);
